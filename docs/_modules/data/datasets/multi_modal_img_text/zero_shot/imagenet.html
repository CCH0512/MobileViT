        root = self.root
        ImageFolder.__init__(
            self, root=root, transform=None, target_transform=None, is_valid_file=None
        )

        n_classes = len(list(self.class_to_idx.keys()))

        class_names: List = self.class_names()

        templates = []
        for class_id in range(n_classes):
            templates_class_i = clip_text_template(class_names[class_id].lower())
            templates.append(templates_class_i)

        self.text_templates = templates


    def __call__(self, img_index: int) -> Tuple[str, List[List[str]], int]:
        """
        :param img_index: Index of the image
        :return: Tuple[str, List[str]]: Tuple containing image path and list of captions
        """

        img_path, target = self.samples[img_index]
        return img_path, self.text_templates, target
